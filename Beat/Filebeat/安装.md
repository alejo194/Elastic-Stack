#### 配置文件 filebeat.yml
```bash
filebeat.inputs:
- type: docker
  combine_partial: true
  containers:
    path: "/var/lib/docker/containers"
    stream: "stderr"
    ids:
      - "*"
  # 这个文件记录日志读取的位置，如果容器重启，可以从记录的位置开始取日志
  registry_file: ${path.data}/registry
  include_lines: ["ERR*", "err*"]            # 只发送包含这些字样的日志
  exclude_lines: ["DBG*","INFO", "WARN*", "warn*", "HTTP/1.1 200", "200", "^\\s+[\\-`('.|_]"]
  multiline.pattern: '^[0-9]{4}-[0-9]{2}-[0-9]{2}'
  multiline.negate: true
  multiline.match: after
  multiline.timeout: 10s
  fields:
    host: mw13
processors:
  - drop_event:
     when:
       and:
         - equals:
             http.response.code: 200
         - equals:
             code: 200
         - equals:
             status: OK
processors:
  - add_docker_metadata: ~
filebeat.config.modules:
  path: ${path.config}/modules.d/*.yml   #配置加载的全局模式
  reload.enabled: false
output.logstash:
  enabled: true
  hosts: ["10.40.201.130:5044"]
output.console:
  enabled: false
  pretty: true
```
